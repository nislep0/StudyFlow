  --
  +email: varchar <<UNIQUE>>

}

class courses {
  +id: UUID <<PK>>
  --
  +user_id: UUID <<FK>>
  +name: varchar
  +description: text?
  +created_at: timestamp
  +updated_at: timestamp
}

  +id: UUID <<PK>>
  --
  +course_id: UUID <<FK>>

  +description: text?
  +status: enum(planned,in_progress,done,overdue)
  +priority: enum(low,medium,high)
  +due_date: date?
  +created_at: timestamp
  +updated_at: timestamp
}
